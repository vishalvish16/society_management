class ComplaintsList extends StatelessWidget {
  final List<Complaint> complaints;

  const ComplaintsList({required this.complaints});

  @override
  Widget build(BuildContext context) {
    return ListView.builder(
      itemCount: complaints.length,
      itemBuilder: (context, index) {
        final complaint = complaints[index];
        return Padding(
          padding: EdgeInsets.all(8),
          child: Container(
            decoration: BoxDecoration(
              color: AppColors.surface,
              borderRadius: BorderRadius.circular(12),
              border: Border.all(color: Color(0xFFE2E8F0)),
            ),
            child: ListTile(
              contentPadding: EdgeInsets.all(16),
              leading: Icon(Icons.category, size: 32, color: AppColors.textMuted),
              title: Text(complaint.title, style: AppTextStyles.bodyMedium),
              subtitle: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: [
                  SizedBox(height: 4),
                  Text('Unit: ${complaint.unit}', style: AppTextStyles.bodySmall),
                  SizedBox(height: 4),
                  Text('Date: ${complaint.date}', style: AppTextStyles.bodySmall),
                ],
              ),
              trailing: Container(
                padding: EdgeInsets.symmetric(horizontal: 8, vertical: 2),
                decoration: BoxDecoration(
                  color: getPriorityColor(complaint.priority),
                  borderRadius: BorderRadius.circular(12),
                ),
                child: Text(
                  complaint.priority,
                  style: AppTextStyles.bodySmall.copyWith(color: Colors.white),
                ),
              ),
              onTap: () => context.go('/complaint/${complaint.id}'),
            ),
          )
              .animate()
              .fadeIn(duration: 300.ms)
              .slideY(begin: 0.2, end: 0),
        );
      },
    );
  }

  Color getPriorityColor(String priority) {
    switch (priority.toLowerCase()) {
      case 'high':
        return AppColors.error;
      case 'medium':
        return AppColors.warning;
      case 'low':
        return AppColors.info;
      default:
        return AppColors.textMuted;
    }
  }
}

class Complaint {

  final String priority;

    required this.id,
    required this.title,
    required this.unit,
    required this.date,
    required this.priority,
  });
}
